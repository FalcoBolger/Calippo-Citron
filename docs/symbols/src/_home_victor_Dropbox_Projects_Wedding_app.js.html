 86 // Create a new Blogpost
 87 
 88 app.post('/article', function (req, res) {
 89 
 90     // This is the entry point for adding new articles to the Blog
 91 
 92     // Modify to implement the Commander Pattern
 93 
 94     if (req.body && req.body.article) {
 95         articles.push(req.body.article);
 96         if (req.body.article.published === "true")  {
 97             res.send({
 98                 status: "ok",
 99                 message: "Article Published"
100             });
101         } else {
102             res.send({
103                 status: "ok",
104                 message: "Article Received"
105             });
106         }
107     } else {
108         res.send({
109             status: "nok",
110             message: "No Articles Received"
111         });
112     }
113 });
